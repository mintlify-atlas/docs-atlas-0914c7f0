---
title: HTML DSL
description: Type-safe HTML DSL for building virtual DOM elements with attributes and event handlers
---

The HTML DSL provides a type-safe API for constructing virtual DOM elements using Effect. All elements and attributes are typed and support message-driven event handling.

## Core Types

### Html

```typescript
type Html = Effect.Effect<VNode | null, never, Dispatch>
```

A virtual DOM element represented as an Effect that produces a VNode. This is the return type of all element constructors.

### TagName

Union type of all valid HTML, SVG, and MathML tag names. Includes standard HTML elements (`div`, `button`, `input`, etc.), SVG elements (`svg`, `path`, `circle`, etc.), and MathML elements (`math`, `mfrac`, etc.).

### Attribute\<Message\>

Tagged union of all valid HTML, SVG, and MathML attributes. See [Attributes](#attributes) for the complete list.

### KeyboardModifiers

```typescript
type KeyboardModifiers = Readonly<{
  shiftKey: boolean
  ctrlKey: boolean
  altKey: boolean
  metaKey: boolean
}>
```

Modifier key state extracted from keyboard events.

## Creating Elements

All element functions are accessed through the `html()` factory function:

```typescript
import { html } from 'foldkit/html'

const { div, button, input, span } = html<Message>()
```

### Standard Elements

Most HTML elements accept attributes and children:

```typescript
const element = div(
  [Class('container'), Id('main')],
  [span([], ['Hello']), button([OnClick(message)], ['Click me'])]
)
```

<ParamField path="attributes" type="ReadonlyArray<Attribute<Message>>" default="[]">
  Array of attributes to apply to the element.
</ParamField>

<ParamField path="children" type="ReadonlyArray<Html | string>" default="[]">
  Child elements or text nodes. Strings are rendered as text nodes.
</ParamField>

### Void Elements

Self-closing elements (like `input`, `img`, `br`) only accept attributes:

```typescript
const selfClosing = input([Type('text'), Value(model.username)])
```

<ParamField path="attributes" type="ReadonlyArray<Attribute<Message>>" default="[]">
  Array of attributes to apply to the element.
</ParamField>

### Keyed Elements

Use keyed elements when you need stable identity for efficient diffing:

```typescript
const { keyed } = html<Message>()

const item = keyed('div')(
  `item-${id}`,
  [Class('item')],
  [content]
)
```

<ParamField path="key" type="string" required>
  Unique string identifier for this element instance. Must be stable across renders.
</ParamField>

<ParamField path="attributes" type="ReadonlyArray<Attribute<Message>>" default="[]">
  Array of attributes to apply to the element (excluding Key).
</ParamField>

<ParamField path="children" type="ReadonlyArray<Html | string>" default="[]">
  Child elements or text nodes.
</ParamField>

### Empty Element

```typescript
const { empty } = html<Message>()

const maybeElement = condition ? div([], ['content']) : empty
```

Returns `null` — useful for conditional rendering.

## Memoization

### createLazy

```typescript
import { createLazy } from 'foldkit/html'

const lazy = createLazy()

const expensiveView = (data: Data): Html =>
  lazy(renderExpensiveView, [data])
```

Creates a memoization slot for a view function. Returns cached VNode if function and arguments are referentially equal (`===`) to the previous call.

<ParamField path="fn" type="(...args: Args) => Html" required>
  View function to memoize.
</ParamField>

<ParamField path="args" type="ReadonlyArray<unknown>" required>
  Arguments to pass to the function. Compared by reference.
</ParamField>

### createKeyedLazy

```typescript
import { createKeyedLazy } from 'foldkit/html'

const lazy = createKeyedLazy()

const items = Array.map(data, item =>
  lazy(item.id, renderItem, [item])
)
```

Creates a keyed memoization map for view functions rendered in a loop. Each key gets its own cache slot.

<ParamField path="key" type="string" required>
  Unique identifier for this cache slot.
</ParamField>

<ParamField path="fn" type="(...args: Args) => Html" required>
  View function to memoize.
</ParamField>

<ParamField path="args" type="ReadonlyArray<unknown>" required>
  Arguments to pass to the function. Compared by reference.
</ParamField>

## Usage Examples

### Basic Element Construction

```typescript
import { html } from 'foldkit/html'
import { m } from 'foldkit/message'
import { S } from 'effect'

const Clicked = m('Clicked')
type Message = typeof Clicked.Type

const view = (): Html => {
  const { div, button, Class, OnClick } = html<Message>()

  return div(
    [Class('container')],
    [
      button(
        [Class('primary'), OnClick(Clicked())],
        ['Click me']
      )
    ]
  )
}
```

### Conditional Rendering

```typescript
const { div, empty } = html<Message>()

const view = (showContent: boolean): Html =>
  div(
    [],
    [
      showContent
        ? div([Class('content')], ['Content'])
        : empty
    ]
  )
```

### Keyed Lists

```typescript
const { div, keyed, Class } = html<Message>()

const items: ReadonlyArray<Item> = [...]

const itemsList = div(
  [Class('list')],
  Array.map(items, item =>
    keyed('div')(
      item.id,
      [Class('item')],
      [item.content]
    )
  )
)
```

### Event Handlers with Data

```typescript
const ItemClicked = m('ItemClicked', { id: S.String })

const { div, OnClick } = html<Message>()

const item = (id: string): Html =>
  div(
    [OnClick(ItemClicked({ id }))],
    ['Click me']
  )
```

### Keyboard Events

```typescript
const KeyPressed = m('KeyPressed', {
  key: S.String,
  shiftKey: S.Boolean
})

const { input, OnKeyDown } = html<Message>()

const field = input([
  OnKeyDown((key, modifiers) =>
    KeyPressed({ key, shiftKey: modifiers.shiftKey })
  )
])
```

### Conditional Event Handling

```typescript
const { input, OnKeyDownPreventDefault } = html<Message>()

const field = input([
  OnKeyDownPreventDefault((key) =>
    key === 'Enter'
      ? Option.some(SubmitForm())
      : Option.none()
  )
])
```
